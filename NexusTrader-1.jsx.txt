// Sparkline — tiny price chart for each instrument
function Spark({ data = [], color = "#00ff88" }) {
  const w = 60, h = 22;
  if (!data || data.length < 2) return <div style={{ width: w, height: h }} />;
  const min = Math.min(...data), max = Math.max(...data);
  const rng = max - min || 0.0001;
  const pts = data.map((v, i) =>
    `${(i / (data.length - 1)) * w},${h - ((v - min) / rng) * (h - 2) - 1}`
  ).join(" ");
  return (
    <svg width={w} height={h}>
      <polyline points={pts} fill="none" stroke={color} strokeWidth="1.5" strokeLinejoin="round" />
    </svg>
  );
}

// RSI gauge bar
function RsiBar({ value = 50 }) {
  const color = value <= 30 ? "#00ff88" : value >= 70 ? "#ff3355" : value >= 55 ? "#ff9944" : "#4488aa";
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 6 }}>
      <div style={{ flex: 1, height: 3, background: "#0d1e32", borderRadius: 2 }}>
        <div style={{ width: `${value}%`, height: "100%", background: color, borderRadius: 2, transition: "width 1s" }} />
      </div>
      <span style={{ fontSize: 10, color, fontFamily: "monospace", minWidth: 28 }}>{value}</span>
    </div>
  );
}

// Loading skeleton
function Skeleton() {
  return (
    <div style={{ padding: 12 }}>
      {[1,2,3,4].map(i => (
        <div key={i} style={{
          background: "#080f1e", border: "1px solid #0d1e32",
          borderRadius: 8, padding: 14, marginBottom: 8, height: 88,
          animation: "pulse 1.5s infinite",
        }} />
      ))}
    </div>
  );
}

// Connection status badge
function StatusBadge({ status }) {
  const cfg = {
    connected:  { color: "#00ff88", label: "LIVE",    dot: true },
    loading:    { color: "#ffcc00", label: "LOADING", dot: false },
    error:      { color: "#ff3355", label: "OFFLINE", dot: false },
  }[status] || { color: "#334466", label: "---", dot: false };

  return (
    <div style={{ display: "flex", alignItems: "center", gap: 5 }}>
      {cfg.dot && (
        <div style={{
          width: 6, height: 6, borderRadius: "50%",
          background: cfg.color, animation: "pulse 2s infinite",
        }} />
      )}
      <span style={{ fontSize: 9, color: cfg.color, fontFamily: "monospace", letterSpacing: 1 }}>
        {cfg.label}
      </span>
    </div>
  );
}

// ── AI CHAT ────────────────────────────────────────────────────
// This calls the Claude API directly from the browser.
// The API key is handled by the Claude artifacts environment.

async function callNexusAI(messages, marketData) {
  const snapshot = Object.entries(marketData)
    .map(([sym, d]) => `${sym}: Price=${d.price} RSI=${d.rsi} Signal=${d.signal}(${d.strength}) Change=${d.changePct}%`)
    .join("\n");

  const system = `You are NEXUS, an elite AI trading analyst bot.
Live market data right now:
${snapshot}

Rules:
- Max 130 words. Be sharp and direct.
- Use real numbers from the data above.
- Format: BUY/SELL/HOLD + reason + stop loss + take profit when giving signals.
- Always mention risk management.
- Cover forex, gold, ETFs.
- Never promise guaranteed returns.`;

  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system,
      messages: messages.map(m => ({ role: m.role, content: m.content })),
    }),
  });
  const data = await res.json();
  return data.content?.[0]?.text || "Signal unavailable.";
}

                        border: "1px solid #00336622",
                        display: "flex", alignItems: "center", justifyContent: "center",
                        fontSize: 12, color: "#0066cc",
                      }}>⬡</div>
                    )}
                    <div style={{
                      maxWidth: "83%",
                      background: m.role === "user" ? "#091828" : "#080f1e",
                      border: "1px solid #0d1e32",
                      borderRadius: m.role === "user" ? "12px 12px 2px 12px" : "12px 12px 12px 2px",
                      padding: "10px 13px",
                      fontSize: 13, color: "#8899bb", lineHeight: 1.65,
                      whiteSpace: "pre-wrap",
                    }}>
                      {m.content}
                    </div>
                  </div>
                ))}
                {aiLoading && (
                  <div style={{ display: "flex", gap: 8, marginBottom: 12 }}>
                    <div style={{
                      width: 26, height: 26, borderRadius: 6,
                      background: "linear-gradient(135deg,#001840,#002060)",
                      border: "1px solid #00336622",
                      display: "flex", alignItems: "center", justifyContent: "center",
                      fontSize: 12, color: "#0066cc",
                    }}>⬡</div>
                    <div style={{
                      background: "#080f1e", border: "1px solid #0d1e32",
                      borderRadius: "12px 12px 12px 2px", padding: "12px 16px",
                      display: "flex", gap: 5, alignItems: "center",
                    }}>
                      {[0,1,2].map(j => (
                        <div key={j} style={{
                          width: 5, height: 5, borderRadius: "50%", background: "#0066cc",
                          animation: `pulse 1.2s ${j * 0.2}s infinite`,
                        }} />
                      ))}
                    </div>
                  </div>
                )}
                <div ref={chatEnd} />
              </div>

              {/* Input */}
              <div style={{
                padding: "8px 12px 12px",
                background: "#060d18", borderTop: "1px solid #0a1828",
                display: "flex", gap: 8,
              }}>
                <input
                  value={input}
                  onChange={e => setInput(e.target.value)}
                  onKeyDown={e => e.key === "Enter" && sendMessage()}
                  placeholder="Ask NEXUS anything..."
                  style={{
                    flex: 1, background: "#080f1e",
                    border: "1px solid #0d1e32", borderRadius: 8,
                    color: "#8899bb", padding: "10px 14px",
                    fontSize: 13, fontFamily: "'Rajdhani',sans-serif",
                  }}
                />
                <button onClick={sendMessage} disabled={aiLoading || !input.trim()} style={{
                  background: aiLoading || !input.trim() ? "#0a1628" : "linear-gradient(135deg,#003388,#0055aa)",
                  border: "1px solid #003388", borderRadius: 8,
                  color: aiLoading || !input.trim() ? "#1e3050" : "#fff",
                  padding: "10px 16px", fontSize: 16,
                  cursor: aiLoading || !input.trim() ? "default" : "pointer",
                  transition: "all 0.2s", fontFamily: "monospace",
                }}>▶</button>
              </div>
            </div>
          )}
        </div>

        {/* FOOTER */}
        <div style={{
          borderTop: "1px solid #0a1828", padding: "6px 16px",
          display: "flex", justifyContent: "space-between",
          fontSize: 8, color: "#0f1f30", fontFamily: "monospace",
        }}>
          <span>NEXUS v2.0 · REAL DATA</span>
          <span>⚠ NOT FINANCIAL ADVICE</span>
          <span>API: {API_BASE.replace("http://","")}</span>
        </div>
      </div>
    </>
  );
