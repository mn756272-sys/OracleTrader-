// ── HELPERS ───────────────────────────────────────
function genCandles(base, vol, count = 80) {
  let p = base;
  return Array.from({ length: count }, () => {
    const o = p, m = (Math.random() - 0.48) * vol * 12;
    const h = o + Math.random() * vol * 6, l = o - Math.random() * vol * 5, c = o + m;
    p = c;
    return { o, h: Math.max(h, o, c), l: Math.min(l, o, c), c };
  });
}

function computeRSI(candles, period = 14) {
  const closes = candles.map(c => c.c);
  if (closes.length < period + 1) return 50;
  let g = 0, l = 0;
  for (let i = closes.length - period; i < closes.length; i++) {
    const d = closes[i] - closes[i - 1];
    if (d > 0) g += d; else l -= d;
  }
  const ag = g / period, al = l / period;
  return al === 0 ? 100 : parseFloat((100 - 100 / (1 + ag / al)).toFixed(1));
}

function toHeikinAshi(candles) {
  const ha = [];
  candles.forEach((c, i) => {
    const haC = (c.o + c.h + c.l + c.c) / 4;
    const haO = i === 0 ? (c.o + c.c) / 2 : (ha[i - 1].o + ha[i - 1].c) / 2;
    ha.push({ o: haO, h: Math.max(c.h, haO, haC), l: Math.min(c.l, haO, haC), c: haC });
  });
  return ha;
}

function findSwings(candles, lb = 4) {
  const highs = [], lows = [];
  for (let i = lb; i < candles.length - lb; i++) {
    if (candles.slice(i - lb, i).every(c => c.h <= candles[i].h) &&
        candles.slice(i + 1, i + lb + 1).every(c => c.h <= candles[i].h))
      highs.push({ idx: i, price: candles[i].h });
    if (candles.slice(i - lb, i).every(c => c.l >= candles[i].l) &&
        candles.slice(i + 1, i + lb + 1).every(c => c.l >= candles[i].l))
      lows.push({ idx: i, price: candles[i].l });
  }
  return { highs, lows };
}

function classifyStructure(highs, lows) {
  return {
    labeledHighs: highs.map((s, i) => ({
      ...s,
      label: i === 0 ? "SH" : s.price > highs[i - 1].price ? "HH" : "LH",
      color: i === 0 ? "#aabbdd" : s.price > highs[i - 1].price ? "#00ff88" : "#ff4466",
    })),
    labeledLows: lows.map((s, i) => ({
      ...s,
      label: i === 0 ? "SL" : s.price > lows[i - 1].price ? "HL" : "LL",
      color: i === 0 ? "#aabbdd" : s.price > lows[i - 1].price ? "#00ff88" : "#ff4466",
    })),
  };
}

function getMarketBias(lH, lL) {
  const bull = lH.slice(-3).filter(h => h.label === "HH").length + lL.slice(-3).filter(l => l.label === "HL").length;
  const bear = lH.slice(-3).filter(h => h.label === "LH").length + lL.slice(-3).filter(l => l.label === "LL").length;
  if (bull > bear) return { bias: "BULLISH", color: "#00ff88" };
  if (bear > bull) return { bias: "BEARISH", color: "#ff4466" };
  return { bias: "RANGING", color: "#ffcc00" };
}

function getTL(pts, count) {
  if (pts.length < 2) return null;
  const [p1, p2] = pts.slice(-2);
  const slope = (p2.price - p1.price) / (p2.idx - p1.idx);
  return { x1: p1.idx, y1: p1.price, x2: count - 1, y2: p2.price + slope * (count - 1 - p2.idx) };
}

function computeFib(high, low) {
  const d = high - low;
  return [
    { price: low,        label: "0%",   color: "#ff4466" },
    { price: low+d*.382, label: "38.2%",color: "#ffcc00" },
    { price: low+d*.5,   label: "50%",  color: "#aabbdd" },
    { price: low+d*.618, label: "61.8%",color: "#00aaff" },
    { price: high,       label: "100%", color: "#00ff88" },
  ];
}

function getSignal(sym, candles, rsi, spiked) {
  const cfg = INSTRUMENTS[sym];
  const closes = candles.map(c => c.c);
  const trend = closes.at(-1) > closes.at(-6) ? "UP" : "DOWN";
  if (cfg?.spike) {
    if (cfg.spike.dir === "UP") {
      if (spiked) return { type:"SELL", strength:"STRONG",  color:"#ff4466", note:"Spike — fade it" };
      if (rsi<=35)  return { type:"BUY",  strength:"STRONG",  color:"#00ff88", note:"Oversold — spike incoming" };
      return       { type:"HOLD", strength:"NEUTRAL", color:"#ffcc00", note:"Wait for setup" };
    }
    if (spiked)   return { type:"BUY",  strength:"STRONG",  color:"#00ff88", note:"Crash — bounce play" };
    if (rsi>=65)  return { type:"SELL", strength:"STRONG",  color:"#ff4466", note:"Overbought — crash incoming" };
    return         { type:"HOLD", strength:"NEUTRAL", color:"#ffcc00", note:"Wait" };
  }
  if (rsi <= 30) return { type:"BUY",  strength:"STRONG",   color:"#00ff88", note:"RSI Oversold" };
  if (rsi >= 70) return { type:"SELL", strength:"STRONG",   color:"#ff4466", note:"RSI Overbought" };
  if (rsi <= 42 && trend === "UP")   return { type:"BUY",  strength:"MODERATE", color:"#44cc88", note:"Bullish momentum" };
  if (rsi >= 60 && trend === "DOWN") return { type:"SELL", strength:"MODERATE", color:"#ff6644", note:"Bearish momentum" };
  return { type:"HOLD", strength:"NEUTRAL", color:"#ffcc00", note:"No clear signal" };
}

// ── MARKET DATA HOOK ──────────────────────────────
function useMarketData() {
  const [data, setData] = useState(() => {
    const d = {};
    Object.entries(INSTRUMENTS).forEach(([k, cfg]) => {
      const candles = genCandles(cfg.base, cfg.vol);
      d[k] = { price: candles.at(-1).c, changePct: 0, candles,
               history: candles.map(c => c.c).slice(-40),
               spiked: false, type: cfg.type };
    });
    return d;
  });
  const [isLive, setIsLive] = useState(false);
  const [status, setStatus] = useState("sim");

  useEffect(() => {
    const iv = setInterval(() => {
      setData(prev => {
        const next = { ...prev };
        Object.entries(INSTRUMENTS).forEach(([sym, cfg]) => {
          const d = next[sym];
          let delta = (Math.random() - 0.499) * cfg.vol * 8;
          let spiked = false;
          if (cfg.spike) {
            delta += cfg.spike.dir === "UP" ? -cfg.vol * 0.5 : cfg.vol * 0.5;
            if (Math.random() < 1 / cfg.spike.freq) {
              delta += cfg.spike.dir === "UP" ? cfg.spike.mag : -cfg.spike.mag;
              spiked = true;
            }
          }
          const last = d.candles.at(-1);
          const nc = parseFloat(Math.max(cfg.base * 0.5, last.c + delta).toFixed(cfg.type === "FOREX" ? 4 : 2));
          const candles = [...d.candles.slice(1), { o: last.c, h: Math.max(last.c, nc) + Math.abs(delta) * 0.2, l: Math.min(last.c, nc) - Math.abs(delta) * 0.2, c: nc }];
          const history = candles.map(c => c.c).slice(-40);
          const changePct = parseFloat(((nc - history[0]) / history[0] * 100).toFixed(3));
          next[sym] = { ...d, price: nc, changePct, candles, history, spiked };
        });
        return next;
      });
    }, 1600);
    return () => clearInterval(iv);
  }, []);

  const fetchLive = useCallback(async () => {
    try {
      const res = await fetch("http://localhost:8000/prices", { signal: AbortSignal.timeout(6000) });
      if (!res.ok) throw new Error();
      const json = await res.json();
      if (Object.keys(json).length > 0) { setData(json); setIsLive(true); setStatus("live"); }
    } catch { setIsLive(false); setStatus("sim"); }
  }, []);

  useEffect(() => { fetchLive(); const iv = setInterval(fetchLive, 30000); return () => clearInterval(iv); }, [fetchLive]);
  return { data, isLive, status, retry: fetchLive };
}

function RsiBar({ value }) {
  const c = value <= 30 ? "#00ff88" : value >= 70 ? "#ff4466" : "#6688aa";
  return (
    <div style={{ display:"flex", alignItems:"center", gap:6 }}>
      <div style={{ flex:1, height:4, background:"#1a2235", borderRadius:2, overflow:"hidden" }}>
        <div style={{ width:`${value}%`, height:"100%", background:c, borderRadius:2, transition:"width 0.5s" }} />
      </div>
      <span style={{ fontSize:11, color:c, fontFamily:"monospace", minWidth:28 }}>{value}</span>
    </div>
  );
}

function Spark({ data, color, spiked }) {
  const w = 60, h = 22;
  const min = Math.min(...data), max = Math.max(...data), range = max - min || 0.0001;
  const pts = data.map((v, i) => `${(i / (data.length - 1)) * w},${h - ((v - min) / range) * (h - 2) + 1}`).join(" ");
  return (
    <svg width={w} height={h} style={{ display:"block" }}>
      {spiked && <rect width={w} height={h} fill={color + "15"} rx="2" />}
      <polyline points={pts} fill="none" stroke={color} strokeWidth={spiked ? 2 : 1.5} strokeLinejoin="round" />
    </svg>
  );
}

// ── FULL CHART ────────────────────────────────────
function OracleChart({ sym, instrument }) {
  const [show, setShow] = useState({ trend:true, sr:true, struct:true, fib:false, ha:false, vp:false });
  const candles = instrument?.candles || [];
  if (candles.length < 10) return <div style={{ textAlign:"center", padding:30, color:"#334466", fontFamily:"monospace" }}>Loading chart...</div>;

  const display = show.ha ? toHeikinAshi(candles) : candles;
  const W = 360, CH = 220, RH = 48, PL = 50, PR = 8, PT = 8, CW = W - PL - PR;
  const allP = display.flatMap(c => [c.h, c.l]);
  const minP = Math.min(...allP), maxP = Math.max(...allP), rng = maxP - minP || 0.0001;
  const bW = CW / candles.length;
  const px = i => PL + i * bW + bW / 2;
  const py = v => PT + CH - ((v - minP) / rng) * CH;
  const fmt = v => instrument?.type === "FOREX" ? parseFloat(v).toFixed(4) : parseFloat(v).toFixed(1);

  const { highs, lows } = findSwings(candles);
  const { labeledHighs, labeledLows } = classifyStructure(highs, lows);
  const bias = getMarketBias(labeledHighs, labeledLows);
  const upTL = getTL(lows.slice(-4), candles.length);
  const dwTL = getTL(highs.slice(-4), candles.length);
  const fibH = highs.length ? Math.max(...highs.slice(-2).map(h => h.price)) : maxP;
  const fibL = lows.length  ? Math.min(...lows.slice(-2).map(l => l.price))  : minP;
  const fibs = computeFib(fibH, fibL);

  const rsiArr = candles.map((_, i) => {
    if (i < 14) return 50;
    const sl = candles.slice(i - 14, i + 1).map(c => c.c);
    let g = 0, l = 0;
    for (let j = 1; j < sl.length; j++) { const d = sl[j] - sl[j - 1]; if (d > 0) g += d; else l -= d; }
    const ag = g / 14, al = l / 14;
    return al === 0 ? 100 : parseFloat((100 - 100 / (1 + ag / al)).toFixed(1));
  });
  const rsiPts = rsiArr.map((v, i) => `${px(i)},${PT + CH + RH + 2 - ((v / 100) * (RH - 4))}`).join(" ");
  const lastRSI = rsiArr.at(-1);
  const rsiC = lastRSI <= 30 ? "#00ff88" : lastRSI >= 70 ? "#ff4466" : "#cc77ff";

  const tcoords = line => line ? { x1:px(line.x1), y1:py(line.y1), x2:px(line.x2), y2:py(line.y2) } : null;
  const upC = tcoords(upTL), dwC = tcoords(dwTL);
  const totalH = PT + CH + RH + 20;
  const curY = py(instrument.price);
  const up = instrument.changePct >= 0;

  const TOGGLES = [
    { k:"ha",     label:"HA",    color:"#ffcc00" },
    { k:"trend",  label:"Trend", color:"#9944ff" },
    { k:"sr",     label:"S/R",   color:"#ffcc00" },
    { k:"struct", label:"HH/LL", color:"#00aaff" },
    { k:"fib",    label:"Fib",   color:"#00ff88" },
  ];

  return (
    <div style={{ background:"#06101c", borderRadius:10, overflow:"hidden", marginBottom:12 }}>
      <div style={{ display:"flex", gap:4, padding:"8px 10px", borderBottom:"1px solid #0d1e32", flexWrap:"wrap" }}>
        {TOGGLES.map(b => (
          <button key={b.k} onClick={() => setShow(p => ({ ...p, [b.k]: !p[b.k] }))} style={{
            background: show[b.k] ? b.color + "22" : "#080f1e",
            border: `1px solid ${show[b.k] ? b.color + "66" : "#0d1e32"}`,
            borderRadius:20, color: show[b.k] ? b.color : "#334466",
            fontSize:10, padding:"3px 10px", cursor:"pointer", fontFamily:"monospace",
          }}>{b.label}</button>
        ))}
        <div style={{ marginLeft:"auto", display:"flex", alignItems:"center", gap:5 }}>
          <div style={{ width:6, height:6, borderRadius:"50%", background:bias.color }} />
          <span style={{ fontSize:10, color:bias.color, fontFamily:"monospace" }}>{bias.bias}</span>
        </div>
      </div>

      <svg width="100%" viewBox={`0 0 ${W} ${totalH}`} style={{ display:"block" }}>
        <defs><clipPath id="ch"><rect x={PL} y={PT} width={CW} height={CH} /></clipPath></defs>
        <rect x={PL} y={PT} width={CW} height={CH} fill="#04080f" rx="2" />

        {/* Grid */}
        {[0, .25, .5, .75, 1].map((f, i) => {
          const pv = maxP - f * rng, yp = PT + f * CH;
          return (<g key={i}>
            <line x1={PL} y1={yp} x2={PL + CW} y2={yp} stroke="#0a1830" strokeWidth="0.5" strokeDasharray="3,6" />
            <text x={PL - 3} y={yp + 3} textAnchor="end" fill="#223344" fontSize="7" fontFamily="monospace">{fmt(pv)}</text>
          </g>);
        })}

        {/* Fibonacci */}
        {show.fib && fibs.map((fib, i) => {
          const yf = py(fib.price);
          if (yf < PT || yf > PT + CH) return null;
          return (<g key={i}>
            <line x1={PL} y1={yf} x2={PL + CW} y2={yf} stroke={fib.color} strokeWidth="0.6" strokeDasharray="2,5" opacity="0.5" />
            <text x={PL + CW - 2} y={yf - 1} textAnchor="end" fill={fib.color} fontSize="6" fontFamily="monospace" opacity="0.8">{fib.label}</text>
          </g>);
        })}

        {/* Support / Resistance */}
        {show.sr && [...highs.slice(-3).map(h => ({ price:h.price, type:"R" })), ...lows.slice(-3).map(l => ({ price:l.price, type:"S" }))].map((sr, i) => {
          const ys = py(sr.price);
          if (ys < PT || ys > PT + CH) return null;
          const col = sr.type === "R" ? "#ff4466" : "#00ff88";
          return (<g key={i}>
            <line x1={PL} y1={ys} x2={PL + CW} y2={ys} stroke={col} strokeWidth="0.8" strokeDasharray="4,4" opacity="0.7" />
            <rect x={PL} y={ys - 7} width={14} height={8} fill={col + "22"} rx="1" />
            <text x={PL + 7} y={ys - 1} textAnchor="middle" fill={col} fontSize="6" fontFamily="monospace" fontWeight="bold">{sr.type}</text>
          </g>);
        })}

        {/* Trend lines */}
        {show.trend && upC && <g clipPath="url(#ch)"><line x1={upC.x1} y1={upC.y1} x2={upC.x2} y2={upC.y2} stroke="#00ff88" strokeWidth="1.2" strokeDasharray="5,3" opacity="0.8" /></g>}

        {show.trend && upC && upC.y2 > PT && upC.y2 < PT + CH && <text x={upC.x2 + 2} y={upC.y2} fill="#00ff88" fontSize="7" fontFamily="monospace">↑UPT</text>}

        {/* Candles */}
        <g clipPath="url(#ch)">
          {display.map((c, i) => {
            const x = px(i), isUp = c.c >= c.o;
            const col = instrument.spiked && i === candles.length - 1 ? "#9944ff"
              : show.ha ? (isUp ? "#00ff88" : "#ff4466")
              : (isUp ? "#00cc66" : "#ff3355");
            const bT = py(Math.max(c.o, c.c)), bB = py(Math.min(c.o, c.c));
            return (<g key={i}>
              <line x1={x} y1={py(c.h)} x2={x} y2={py(c.l)} stroke={col} strokeWidth="0.7" />
              <rect x={x - bW * .38} y={bT} width={bW * .76} height={Math.max(bB - bT, 1)}
                fill={show.ha ? col : col + "55"} stroke={col} strokeWidth="0.7" rx="0.3" />
            </g>);
          })}
        </g>

        {/* HH/HL/LH/LL */}
        {show.struct && labeledHighs.slice(-6).map((h, i) => {
          const xH = px(h.idx), yH = py(h.price) - 10;
          if (xH < PL || xH > PL + CW || yH < PT - 2) return null;
          return (<g key={i}>
            <circle cx={xH} cy={py(h.price)} r="2.5" fill={h.color} opacity="0.9" />
            <rect x={xH - 8} y={yH - 7} width={16} height={8} fill="#04080f" opacity="0.85" rx="2" />
            <text x={xH} y={yH} textAnchor="middle" fill={h.color} fontSize="7" fontFamily="monospace" fontWeight="bold">{h.label}</text>
          </g>);
        })}
        {show.struct && labeledLows.slice(-6).map((l, i) => {
          const xL = px(l.idx), yL = py(l.price) + 16;
          if (xL < PL || xL > PL + CW || yL > PT + CH + 4) return null;
          return (<g key={i}>
            <circle cx={xL} cy={py(l.price)} r="2.5" fill={l.color} opacity="0.9" />
            <rect x={xL - 8} y={yL - 7} width={16} height={8} fill="#04080f" opacity="0.85" rx="2" />
            <text x={xL} y={yL} textAnchor="middle" fill={l.color} fontSize="7" fontFamily="monospace" fontWeight="bold">{l.label}</text>
          </g>);
        })}

        {/* Current price line */}
        {(() => {
          const col = up ? "#00ff88" : "#ff4466";
          return (<g>
            <line x1={PL} y1={curY} x2={PL + CW} y2={curY} stroke={col} strokeWidth="0.6" strokeDasharray="2,4" opacity="0.9" />
            <rect x={PL + CW} y={curY - 6} width={50} height={12} fill={col + "33"} rx="2" />
            <text x={PL + CW + 2} y={curY + 3} fill={col} fontSize="7" fontFamily="monospace">{fmt(instrument.price)}</text>
          </g>);
        })()}

        {/* RSI panel */}
        <line x1={PL} y1={PT + CH + 4} x2={PL + CW} y2={PT + CH + 4} stroke="#0d1e32" strokeWidth="1" />
        <rect x={PL} y={PT + CH + 4} width={CW} height={RH} fill="#030710" opacity="0.8" />
        <text x={PL - 3} y={PT + CH + 10} textAnchor="end" fill="#334466" fontSize="6" fontFamily="monospace">RSI</text>
        <line x1={PL} y1={PT + CH + 4 + RH * .3} x2={PL + CW} y2={PT + CH + 4 + RH * .3} stroke="#ff446614" strokeWidth="0.5" />
        <line x1={PL} y1={PT + CH + 4 + RH * .7} x2={PL + CW} y2={PT + CH + 4 + RH * .7} stroke="#00ff8814" strokeWidth="0.5" />
        <polyline points={rsiPts} fill="none" stroke="#cc77ff" strokeWidth="1.2" opacity="0.9" />
        <text x={PL - 3} y={PT + CH + RH} textAnchor="end" fill={rsiC} fontSize="7" fontFamily="monospace">{lastRSI}</text>
      </svg>

      {show.struct && (
        <div style={{ display:"flex", gap:8, padding:"6px 10px", borderTop:"1px solid #0d1e32", flexWrap:"wrap" }}>
          {[["HH","Higher High","#00ff88"],["HL","Higher Low","#00ff88"],["LH","Lower High","#ff4466"],["LL","Lower Low","#ff4466"]].map(([label, desc, color]) => (
            <div key={label} style={{ display:"flex", alignItems:"center", gap:4 }}>
              <div style={{ width:6, height:6, borderRadius:"50%", background:color }} />
              <span style={{ fontSize:9, color:"#334466", fontFamily:"monospace" }}>{label}={desc}</span>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}

async function askOracle(messages, snapshot) {
  const sys = `You are ORACLE, an elite AI trading analyst. You see all — technical analysis, market structure (HH/HL/LH/LL), Fibonacci, RSI, MACD, Boom/Crash synthetic indices, and crypto.
Live market: ${snapshot}
Rules: Max 130 words. Sharp and authoritative. Give entry/SL/TP. Never guarantee returns. Risk 1-2% max.`;
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST", headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ model:"claude-sonnet-4-20250514", max_tokens:1000, system:sys,
      messages: messages.map(m => ({ role:m.role, content:m.content })) }),
  });
  const d = await res.json();
  return d.content?.[0]?.text || "Oracle signal unavailable.";
}

                  <span>RISK PER TRADE</span><span style={{ color:"#9944ff" }}>{rPct}% = ${rAmt.toFixed(2)}</span>
                </div>
                <input type="range" min="0.5" max="5" step="0.5" value={rPct} onChange={e => setRPct(e.target.value)} />
              </Card>
              <Card style={{ marginBottom:12 }}>
                <SectionTitle>TRADE SETUP</SectionTitle>
                {[{ label:"ENTRY PRICE", val:rEntry, set:setREntry, ph:"1.0842" }, { label:"STOP LOSS", val:rSL, set:setRSL, ph:"1.0820" }, { label:"TAKE PROFIT", val:rTP, set:setRTP, ph:"1.0880" }].map(f => (
                  <div key={f.label} style={{ marginBottom:12 }}>
                    <div style={{ fontSize:9, color:"#334466", fontFamily:"monospace", marginBottom:4 }}>{f.label}</div>
                    <input type="number" value={f.val} onChange={e => f.set(e.target.value)} placeholder={f.ph} style={S} />
                  </div>
                ))}
              </Card>
              <Card>
                <SectionTitle>RESULTS</SectionTitle>
                {[{ label:"Risk Amount", val:rAmt > 0 ? `$${rAmt.toFixed(2)}` : "—", color:"#ff4466" }, { label:"Potential Profit", val:potProfit > 0 ? `$${potProfit}` : "—", color:"#00ff88" }, { label:"Lot Size", val:lotSize > 0 ? `${lotSize} lots` : "—", color:"#00aaff" }, { label:"R/R Ratio", val:rrRatio > 0 ? `1:${rrRatio}` : "—", color:rrRatio >= 2 ? "#00ff88" : rrRatio >= 1.5 ? "#ffcc00" : "#ff9944" }].map(({ label, val, color }) => (
                  <div key={label} style={{ display:"flex", justifyContent:"space-between", padding:"8px 0", borderBottom:"1px solid #06101c" }}>
                    <span style={{ fontSize:13, color:"#667799" }}>{label}</span>
                    <span style={{ fontSize:13, color, fontFamily:"monospace", fontWeight:"bold" }}>{val}</span>
                  </div>
                ))}
              </Card>
            </div>
          )}

          {/* ═══ SESSIONS ═══ */}
          {tab === "sessions" && (
            <div style={{ overflowY:"auto", padding:12 }}>
              <div style={{ fontSize:22, fontWeight:900, color:"#ddeeff", marginBottom:4 }}>🌍 Market Sessions</div>
              <div style={{ fontSize:11, color:"#334466", fontFamily:"monospace", marginBottom:16 }}>UTC {clock.toUTCString().slice(17, 22)} · Local {clock.toLocaleTimeString([], { hour:"2-digit", minute:"2-digit" })}</div>
              {SESSIONS.map(s => {
                const open = sessionOpen(s.open, s.close);
                return (
                  <div key={s.name} style={{ background:open ? s.color + "10" : "#080f1e", border:`1px solid ${open ? s.color + "44" : "#0d1e32"}`, borderRadius:10, padding:"14px", marginBottom:10 }}>
                    <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginBottom:8 }}>
                      <div style={{ display:"flex", gap:10, alignItems:"center" }}>
                        <span style={{ fontSize:24 }}>{s.flag}</span>
                        <div>
                          <div style={{ fontSize:15, fontWeight:700, color:open ? s.color : "#667799" }}>{s.name}</div>
                          <div style={{ fontSize:10, color:"#334466", fontFamily:"monospace" }}>{s.open}:00 – {s.close}:00 UTC</div>
                        </div>
                      </div>
                      <div style={{ background:open ? s.color + "22" : "#0d1e32", border:`1px solid ${open ? s.color + "55" : "#1e3050"}`, borderRadius:4, padding:"4px 12px", display:"flex", alignItems:"center", gap:5 }}>
                        {open && <div style={{ width:5, height:5, borderRadius:"50%", background:s.color, animation:"pulse 2s infinite" }} />}
                        <span style={{ fontSize:11, color:open ? s.color : "#334466", fontFamily:"monospace" }}>{open ? "OPEN" : "CLOSED"}</span>
                      </div>
                    </div>
                    <div style={{ display:"flex", gap:6, flexWrap:"wrap" }}>
                      {s.pairs.map(p => <span key={p} style={{ background:s.color + "15", border:`1px solid ${s.color}33`, borderRadius:4, padding:"3px 10px", fontSize:11, color:s.color, fontFamily:"monospace" }}>{p}</span>)}
                    </div>
                  </div>
                );
              })}
            </div>
          )}

          {/* ═══ STATS ═══ */}
          {tab === "stats" && (
            <div style={{ overflowY:"auto", padding:12 }}>
              <div style={{ fontSize:22, fontWeight:900, color:"#ddeeff", marginBottom:4 }}>📈 Performance</div>
              <div style={{ fontSize:11, color:"#334466", fontFamily:"monospace", marginBottom:14 }}>Your trading statistics</div>
              <div style={{ display:"grid", gridTemplateColumns:"1fr 1fr", gap:8, marginBottom:14 }}>
                {[{ label:"Win Rate", val:`${winRate}%`, color:winRate >= 60 ? "#00ff88" : winRate >= 50 ? "#ffcc00" : "#ff4466", sub:`${wins}W · ${losses}L` }, { label:"Total R", val:`${totalR >= 0 ? "+" : ""}${totalR.toFixed(1)}R`, color:totalR >= 0 ? "#00ff88" : "#ff4466", sub:`${trades.length} trades` }, { label:"Avg R", val:`${trades.length ? (totalR / trades.length).toFixed(2) : 0}R`, color:"#cc77ff", sub:"Per trade" }, { label:"Best", val:`+${Math.max(...trades.map(t => t.rr))}R`, color:"#00ff88", sub:"Best trade" }].map(s => (
                  <div key={s.label} style={{ background:"#080f1e", border:"1px solid #0d1e32", borderRadius:10, padding:"14px 12px" }}>
                    <div style={{ fontSize:9, color:"#334466", fontFamily:"monospace", marginBottom:4 }}>{s.label}</div>
                    <div style={{ fontSize:22, fontWeight:900, color:s.color, fontFamily:"monospace" }}>{s.val}</div>
                    <div style={{ fontSize:10, color:"#334466", fontFamily:"monospace", marginTop:2 }}>{s.sub}</div>
                  </div>
                ))}
              </div>
              <Card>
                <SectionTitle>WIN / LOSS BREAKDOWN</SectionTitle>
                <div style={{ display:"flex", justifyContent:"space-between", marginBottom:6, fontSize:10, fontFamily:"monospace" }}>
                  <span style={{ color:"#00ff88" }}>W {wins}</span>
                  <span style={{ color:"#334466" }}>{winRate}% win rate</span>
                  <span style={{ color:"#ff4466" }}>L {losses}</span>
                </div>
                <div style={{ height:8, background:"#0d1e32", borderRadius:4, overflow:"hidden" }}>
                  <div style={{ width:`${winRate}%`, height:"100%", background:"linear-gradient(90deg,#00ff88,#44cc88)", borderRadius:4, transition:"width 0.5s" }} />
                </div>
              </Card>
            </div>
          )}

          {/* ═══ SETTINGS ═══ */}
          {tab === "settings" && (
            <div style={{ overflowY:"auto", padding:12 }}>
              <div style={{ fontSize:22, fontWeight:900, color:"#ddeeff", marginBottom:16 }}>⚙ Settings</div>
              <Card style={{ marginBottom:12 }}>
                <SectionTitle>SIGNAL THRESHOLDS</SectionTitle>
                {[{ label:"RSI Oversold Level", key:"rsiLow" }, { label:"RSI Overbought Level", key:"rsiHigh" }].map(f => (
                  <div key={f.key} style={{ display:"flex", justifyContent:"space-between", alignItems:"center", padding:"12px 0", borderBottom:"1px solid #06101c" }}>
                    <span style={{ fontSize:13, color:"#aabbdd", fontWeight:600 }}>{f.label}</span>
                    <input type="number" value={settings[f.key]} onChange={e => setSettings(p => ({ ...p, [f.key]:parseInt(e.target.value) || 0 }))} style={{ ...S, width:80, padding:"6px 10px" }} />
                  </div>
                ))}
              </Card>
              <Card style={{ marginBottom:12 }}>
                <SectionTitle>CHART</SectionTitle>
                <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", padding:"12px 0" }}>
                  <span style={{ fontSize:13, color:"#aabbdd", fontWeight:600 }}>Default Timeframe</span>
                  <select value={settings.tf} onChange={e => setSettings(p => ({ ...p, tf:e.target.value }))} style={{ ...S, width:100, padding:"6px 10px" }}>
                    {["1m", "5m", "15m", "1h", "4h", "1d"].map(o => <option key={o} value={o}>{o}</option>)}
                  </select>
                </div>
              </Card>
              <Card style={{ marginBottom:12 }}>
                <SectionTitle>NOTIFICATIONS</SectionTitle>
                <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", padding:"12px 0" }}>
                  <div>
                    <div style={{ fontSize:13, color:"#aabbdd", fontWeight:600 }}>Push Notifications</div>
                    <div style={{ fontSize:10, color:"#334466", fontFamily:"monospace" }}>Alerts on this device</div>
                  </div>
                  <div onClick={() => setSettings(p => ({ ...p, push:!p.push }))} style={{ width:44, height:24, borderRadius:12, cursor:"pointer", background:settings.push ? "#9944ff" : "#0d1e32", border:`1px solid ${settings.push ? "#9944ff" : "#1e3050"}`, position:"relative", transition:"all 0.3s" }}>
                    <div style={{ width:18, height:18, borderRadius:"50%", background:settings.push ? "#fff" : "#334466", position:"absolute", top:2, left:settings.push ? 22 : 2, transition:"left 0.3s" }} />
                  </div>
                </div>
              </Card>
              <div style={{ background:"#080f1e", border:"1px solid #0d1e32", borderRadius:10, padding:20, textAlign:"center" }}>
                <div style={{ fontSize:56, marginBottom:8 }}>🔮</div>
                <div style={{ fontFamily:"'Share Tech Mono',monospace", fontSize:16, color:BRAND.color, letterSpacing:3 }}>{BRAND.name}</div>
                <div style={{ fontSize:11, color:"#334466", fontFamily:"monospace", marginTop:4 }}>v1.0 · All-Seeing Intelligence</div>
                <div style={{ fontSize:10, color:"#1e3050", fontFamily:"monospace", marginTop:12 }}>⚠ NOT FINANCIAL ADVICE</div>
              </div>
            </div>
          )}

        </div>

        {/* ── FOOTER ── */}
        <div style={{ borderTop:"1px solid #1a0a2e", padding:"7px 16px", display:"flex", justifyContent:"space-between", fontSize:9, color:"#1e3050", fontFamily:"monospace" }}>
          <span>{BRAND.icon} {BRAND.name} v1.0</span>
          <span style={{ color:isLive ? "#00ff8844" : "#ffcc0044" }}>{isLive ? "REAL DATA" : "SIMULATED"}</span>
          <span>{clock.toLocaleTimeString([], { hour:"2-digit", minute:"2-digit" })}</span>
        </div>
      </div>
    </>
  );
}
