function genCandles(base, vol, count=80) {
  let p = base;
  return Array.from({length:count}, () => {
    const o=p, move=(Math.random()-0.48)*vol*12;
    const h=o+Math.random()*vol*6, l=o-Math.random()*vol*5, c=o+move;
    p=c;
    return {o, h:Math.max(h,o,c), l:Math.min(l,o,c), c};
  });
}

function useSimulation() {
  const [sim, setSim] = useState(() => {
    const p = {};
    Object.entries(SIMULATED).forEach(([k,cfg]) => {
      const candles = genCandles(cfg.base, cfg.vol);
      p[k] = { price:candles.at(-1).c, change:0, changePct:0,
               candles, history:candles.map(c=>c.c).slice(-40),
               spiked:false, type:cfg.type, source:"Simulated" };
    });
    return p;
  });

  useEffect(() => {
    const iv = setInterval(() => {
      setSim(prev => {
        const next = {...prev};
        Object.entries(SIMULATED).forEach(([sym,cfg]) => {
          const data = next[sym];
          let delta = (Math.random()-0.499)*cfg.vol*8;
          let spiked = false;
          if (cfg.spike) {
            delta += cfg.spike.dir==="UP" ? -cfg.vol*0.5 : cfg.vol*0.5;
            if (Math.random() < 1/cfg.spike.freq) {
              delta += cfg.spike.dir==="UP" ? cfg.spike.mag : -cfg.spike.mag;
              spiked = true;
            }
          }
          const last = data.candles.at(-1);
          const nc = parseFloat(Math.max(cfg.base*0.5, last.c+delta)
                     .toFixed(cfg.type==="FOREX"?4:2));
          const newCandle = {o:last.c, h:Math.max(last.c,nc)+Math.abs(delta)*0.2,
                             l:Math.min(last.c,nc)-Math.abs(delta)*0.2, c:nc};
          const candles = [...data.candles.slice(1), newCandle];
          const history = candles.map(c=>c.c).slice(-40);
          const open = history[0];
          const change = parseFloat((nc-open).toFixed(4));
          const changePct = parseFloat(((change/open)*100).toFixed(3));
          next[sym] = {...data, price:nc, change, changePct, candles, history, spiked};
        });
        return next;
      });
    }, 1600);
    return () => clearInterval(iv);
  }, []);
  return sim;
}

// ── Technical Analysis ────────────────────────────────────────
function computeRSI(candles, period=14) {
  const closes = candles.map(c=>c.c);
  if (closes.length < period+1) return 50;
  let g=0, l=0;
  for (let i=closes.length-period; i<closes.length; i++) {
    const d = closes[i]-closes[i-1];
    if (d>0) g+=d; else l-=d;
  }
  const ag=g/period, al=l/period;
  if (al===0) return 100;
  return parseFloat((100-100/(1+ag/al)).toFixed(1));
}

function findSwings(candles, lookback=4) {
  const highs=[], lows=[];
  for (let i=lookback; i<candles.length-lookback; i++) {
    if (candles.slice(i-lookback,i).every(c=>c.h<=candles[i].h) &&
        candles.slice(i+1,i+lookback+1).every(c=>c.h<=candles[i].h))
      highs.push({idx:i, price:candles[i].h});
    if (candles.slice(i-lookback,i).every(c=>c.l>=candles[i].l) &&
        candles.slice(i+1,i+lookback+1).every(c=>c.l>=candles[i].l))
      lows.push({idx:i, price:candles[i].l});
  }
  return {highs, lows};
}

function classifyStructure(highs, lows) {
  return {
    labeledHighs: highs.map((s,i) => ({...s,
      label:i===0?"SH":s.price>highs[i-1].price?"HH":"LH",
      color:i===0?"#aabbdd":s.price>highs[i-1].price?"#00ff88":"#ff4466"})),
    labeledLows: lows.map((s,i) => ({...s,
      label:i===0?"SL":s.price>lows[i-1].price?"HL":"LL",
      color:i===0?"#aabbdd":s.price>lows[i-1].price?"#00ff88":"#ff4466"})),
  };
}

function computeFib(high, low) {
  const d = high-low;
  return [
    {level:0,    price:low,       label:"0%",   color:"#ff4466"},
    {level:0.382,price:low+d*.382,label:"38.2%",color:"#ffcc00"},
    {level:0.5,  price:low+d*.5,  label:"50%",  color:"#aabbdd"},
    {level:0.618,price:low+d*.618,label:"61.8%",color:"#00aaff"},
    {level:1,    price:high,      label:"100%", color:"#00ff88"},
  ];
}

function getTrendLine(points, count) {
  if (points.length<2) return null;
  const [p1,p2] = points.slice(-2);
  const slope = (p2.price-p1.price)/(p2.idx-p1.idx);
  return {x1:p1.idx, y1:p1.price, x2:count-1, y2:p2.price+slope*(count-1-p2.idx)};
}

function getMarketBias(lH, lL) {
  const bull = lH.slice(-3).filter(h=>h.label==="HH").length +
               lL.slice(-3).filter(l=>l.label==="HL").length;
  const bear = lH.slice(-3).filter(h=>h.label==="LH").length +
               lL.slice(-3).filter(l=>l.label==="LL").length;
  if (bull>bear) return {bias:"BULLISH",color:"#00ff88"};
  if (bear>bull) return {bias:"BEARISH",color:"#ff4466"};
  return {bias:"RANGING",color:"#ffcc00"};
}

function Spark({data,color,spiked}) {
  const w=64,h=24;
  const min=Math.min(...data),max=Math.max(...data),range=max-min||0.0001;
  const pts=data.map((v,i)=>`${(i/(data.length-1))*w},${h-((v-min)/range)*(h-2)+1}`).join(" ");
  return (
    <svg width={w} height={h} style={{display:"block"}}>
      {spiked&&<rect width={w} height={h} fill={color+"15"} rx="2"/>}
      <polyline points={pts} fill="none" stroke={color} strokeWidth={spiked?2:1.5} strokeLinejoin="round"/>
    </svg>
  );
}
